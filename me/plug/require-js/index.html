require.js测试

juqery.alert.js与jquery.event.js暂时为空，仅为测试使用

引入js/app.js

<script data-main="js/app" src="js/lib/require.js"></script>

在js/app.js中进行路径和目录的配置

requirejs.config({
    //路径
    "baseUrl": "js/lib",
    //目录
    "paths": {
      "app": "../app"
    }
});

//加载主要的 app module 开始执行app
requirejs(["app/main"]);

在app/main.js中进行基于jquery.alert.js与jquery.event.js的操作

define(["jquery", "jquery.alert", "jquery.event"], function($) {
    //确定jquery.alert.js, jquery.event.js 加载完毕.
    $(function() {
        $("#handler-btn").click(function(){
            alert("我是jquery.alert.js, jquery.event.js 加载完毕，弹出的窗口！")
        })
    });
});
